-- This file contains all the necessary SQL queries to create the database schema for the 3D Hub E-Commerce application.
-- It is designed for PostgreSQL and should be executed to set up the database tables.

-- Drop tables if they exist to ensure a clean slate.
-- The order is important due to foreign key constraints.
DROP TABLE IF EXISTS contact_submissions;
DROP TABLE IF EXISTS settings;
DROP TABLE IF EXISTS reviews;
DROP TABLE IF EXISTS order_items;
DROP TABLE IF EXISTS orders;
DROP TABLE IF EXISTS products;
DROP TABLE IF EXISTS categories;
DROP TABLE IF EXISTS users;

-- Users Table
-- Stores user information, including customers and admins.
CREATE TABLE users (
    id VARCHAR(255) PRIMARY KEY,
    name VARCHAR(255) NOT NULL,
    email VARCHAR(255) UNIQUE NOT NULL,
    password VARCHAR(255), -- Can be null for customers who don't create an account
    avatar VARCHAR(255),
    phone VARCHAR(20),
    role VARCHAR(50) NOT NULL CHECK (role IN ('Admin', 'Customer')),
    joined_date TIMESTAMPTZ NOT NULL DEFAULT NOW(),
    created_at TIMESTAMPTZ NOT NULL DEFAULT NOW(),
    updated_at TIMESTAMPTZ NOT NULL DEFAULT NOW()
);

-- Categories Table
-- Stores product categories.
CREATE TABLE categories (
    id VARCHAR(255) PRIMARY KEY,
    name VARCHAR(255) NOT NULL,
    icon VARCHAR(255),
    created_at TIMESTAMPTZ NOT NULL DEFAULT NOW(),
    updated_at TIMESTAMPTZ NOT NULL DEFAULT NOW()
);

-- Products Table
-- Stores all product information.
CREATE TABLE products (
    id VARCHAR(255) PRIMARY KEY,
    name VARCHAR(255) NOT NULL,
    description TEXT,
    price NUMERIC(10, 2) NOT NULL,
    old_price NUMERIC(10, 2),
    stock INTEGER NOT NULL DEFAULT 0,
    category_id VARCHAR(255) REFERENCES categories(id),
    images TEXT[], -- Array of image URLs
    sku VARCHAR(100) UNIQUE NOT NULL,
    enabled BOOLEAN NOT NULL DEFAULT TRUE,
    tags TEXT[],
    weight_kg NUMERIC(10, 2),
    dimensions JSONB, -- e.g., {"length": 30, "width": 30, "height": 2.5}
    created_at TIMESTAMPTZ NOT NULL DEFAULT NOW(),
    updated_at TIMESTAMPTZ NOT NULL DEFAULT NOW()
);
-- Add an index on category_id for faster lookups
CREATE INDEX idx_products_category_id ON products(category_id);


-- Orders Table
-- Stores high-level information about each order.
CREATE TABLE orders (
    id VARCHAR(255) PRIMARY KEY,
    order_date TIMESTAMPTZ NOT NULL DEFAULT NOW(),
    customer_name VARCHAR(255) NOT NULL,
    customer_email VARCHAR(255) NOT NULL,
    customer_phone VARCHAR(20),
    total_amount NUMERIC(10, 2) NOT NULL,
    shipping_address JSONB NOT NULL,
    status VARCHAR(50) NOT NULL CHECK (status IN ('Pending', 'Processing', 'Shipped', 'Delivered', 'Cancelled')),
    user_id VARCHAR(255) REFERENCES users(id) ON DELETE SET NULL, -- Link to a registered user, optional
    shipping_info JSONB, -- e.g., {"carrier": "Express", "trackingNumber": "..."}
    payment_details JSONB,
    created_at TIMESTAMPTZ NOT NULL DEFAULT NOW(),
    updated_at TIMESTAMPTZ NOT NULL DEFAULT NOW()
);
-- Add an index on user_id for faster lookups
CREATE INDEX idx_orders_user_id ON orders(user_id);


-- Order Items Table
-- Stores individual items within an order.
CREATE TABLE order_items (
    id SERIAL PRIMARY KEY,
    order_id VARCHAR(255) NOT NULL REFERENCES orders(id) ON DELETE CASCADE,
    product_id VARCHAR(255) NOT NULL REFERENCES products(id),
    product_name VARCHAR(255) NOT NULL,
    quantity INTEGER NOT NULL,
    price NUMERIC(10, 2) NOT NULL,
    old_price NUMERIC(10, 2),
    image VARCHAR(255),
    created_at TIMESTAMPTZ NOT NULL DEFAULT NOW(),
    updated_at TIMESTAMPTZ NOT NULL DEFAULT NOW()
);
-- Add an index on order_id and product_id for faster lookups
CREATE INDEX idx_order_items_order_id ON order_items(order_id);
CREATE INDEX idx_order_items_product_id ON order_items(product_id);

-- Reviews Table
-- Stores customer reviews for products, linked to a specific order.
CREATE TABLE reviews (
    id SERIAL PRIMARY KEY,
    product_id VARCHAR(255) NOT NULL REFERENCES products(id) ON DELETE CASCADE,
    user_id VARCHAR(255) NOT NULL REFERENCES users(id) ON DELETE CASCADE,
    order_id VARCHAR(255) NOT NULL REFERENCES orders(id) ON DELETE CASCADE,
    rating INTEGER NOT NULL CHECK (rating >= 1 AND rating <= 5),
    comment TEXT,
    created_at TIMESTAMPTZ NOT NULL DEFAULT NOW(),
    -- A user can only review a product once per order
    UNIQUE (product_id, user_id, order_id)
);

-- Add indexes for performance
CREATE INDEX idx_reviews_product_id ON reviews(product_id);
CREATE INDEX idx_reviews_user_id ON reviews(user_id);


-- Settings Table
-- Stores website settings as a single JSONB object for flexibility.
CREATE TABLE settings (
    id INTEGER PRIMARY KEY,
    data JSONB NOT NULL,
    updated_at TIMESTAMPTZ NOT NULL DEFAULT NOW()
);

-- Contact Submissions Table
-- Stores messages sent through the contact form.
CREATE TABLE contact_submissions (
    id SERIAL PRIMARY KEY,
    name VARCHAR(255) NOT NULL,
    email VARCHAR(255) NOT NULL,
    message TEXT NOT NULL,
    status VARCHAR(50) NOT NULL DEFAULT 'New' CHECK (status IN ('New', 'Read', 'Archived')),
    submitted_at TIMESTAMPTZ NOT NULL DEFAULT NOW()
);


-- Note: Initial data (like the default admin user and settings)
-- will be inserted by the seeder script (seeder.js), not here.
-- This keeps the schema definition clean.
